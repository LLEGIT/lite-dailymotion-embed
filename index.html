üìù Usage Examples

Basic HTML

<!-- Simple video embed --> <lite-dailymotion videoid="x7u31wn"></lite-dailymotion> <!-- With title and autoplay disabled --> <lite-dailymotion videoid="x7u31wn" title="My Video Title" autoplay="false"> </lite-dailymotion> <!-- Muted autoplay starting at 1 minute --> <lite-dailymotion videoid="x7u31wn" title="Tutorial Video" mute="true" autoplay="true" starttime="60"> </lite-dailymotion>

JavaScript API

// Get element reference const video = document.querySelector('lite-dailymotion'); // Play the video video.play(); // Get video ID console.log(video.getVideoId()); // Get current state console.log(video.getState()); // 'idle', 'loading', 'loaded', 'error' // Get performance metrics console.log(video.getMetrics()); // Listen to events video.addEventListener('lite-dailymotion-loaded', (e) => { console.log('Video loaded!', e.detail.metrics); }); video.addEventListener('lite-dailymotion-error', (e) => { console.error('Video error:', e.detail.error); });
